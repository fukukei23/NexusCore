204 def _apply_solution_pattern(self, code: str, solution: dict) -> str | None: 

205 # (このメソッドは変更なし) 

206 solution_type = solution.get("type") 

207 if solution_type == "regex_replace": 

208 search_pattern = solution["search"] 

209 replace_template = solution["replace"] 

210 replace_template = re.sub(r'\$(\d)', r'\\\1', replace_template) 

211 return re.sub(search_pattern, replace_template, code, flags=re.DOTALL) 

212 elif solution_type == "add_import": 

213 import_statement = solution["import"] 

214 if import_statement not in code: 

215 return f"{import_statement}\n{code}" 

216 return code 

217 elif solution_type == "regex_replace_with_import": 

218 import_statement = solution["import_statement"] 

219 modified_code = code 

220 if not re.search(fr"^\s*import\s+{re.escape(import_statement.split(' ')[1])}", code, re.MULTILINE): 

221 if import_statement not in modified_code: 

222 modified_code = f"{import_statement}\n{modified_code}" 

223 search_pattern = solution["search"] 

224 replace_template = solution["replace"] 

225 replace_template = re.sub(r'\$(\d)', r'\\\1', replace_template) 

226 return re.sub(search_pattern, replace_template, modified_code, flags=re.DOTALL) 

227 return None

229 def _create_diff(self, original_code: str, modified_code: str, filename: str) -> str: 

230 # (このメソッドは変更なし) 

231 rel_path = os.path.relpath(filename, self.project_path) 

232 filename_for_diff = Path(rel_path).as_posix() 

233 diff = difflib.unified_diff( 

234 original_code.splitlines(keepends=True), 

235 modified_code.splitlines(keepends=True), 

236 fromfile=filename_for_diff, 

237 tofile=filename_for_diff, 

238 ) 

239 return "".join(diff)
