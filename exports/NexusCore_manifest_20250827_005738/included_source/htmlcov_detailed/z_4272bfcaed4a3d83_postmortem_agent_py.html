8import json 

9import re 

10import logging 

11from .base_agent import BaseAgent 

12 

13class PostmortemAgent(BaseAgent): 

14 SYSTEM_PROMPT = """ 

15あなたは、非常に経験豊富なソフトウェア開発者であり、根本原因分析（RCA）の専門家です。 

16あなたの仕事は、他のAIエージェントが解決に失敗したテストエラーの完全なコンテキストを分析し、 

17そのエラーを将来解決するための「知識」をJSON形式で生成することです。 

18""" 

19 

20 def analyze_failure_and_suggest_fkb_entry( 

21 self, 

22 error_log: str, 

23 source_code: str, 

24 test_code: str, 

25 source_file_path: str, 

26 test_file_path: str 

27 ) -> dict | None: 

28 """ 

29 未知のエラーを分析し、fkb_local.jsonに追加すべき新しいエントリを提案する。 

30 成功した場合は辞書を、失敗した場合はNoneを返す。 

31 """ 

32 self.logger.info(f"Analyzing failed test to generate new FKB entry (source={source_file_path}, test={test_file_path})") 

33 

34 # ▼▼▼▼▼ ここからが最重要修正点 ▼▼▼▼▼ 

35 prompt = f""" 

36# 状況 

37我々のAI開発システムが、以下のテストエラーの自己修復に失敗しました。 

38原因は、故障知識ベース（FKB）に、このエラーを解決するためのルールが存在しなかったことです。 

39あなたの任務は、この失敗事例を分析し、FKBに追加すべき新しい知識エントリを1つ生成することです。 

40 

41# 分析対象データ 

42--- 

43## 1. 失敗したテストのエラーログ 

44``` 

45{error_log} 

46``` 

47--- 

48## 2. エラーが発生したソースコード (`{source_file_path}`) 

49```python 

50{source_code} 

51``` 

52--- 

53## 3. 失敗したテストコード (`{test_file_path}`) 

54```python 

55{test_code} 

56``` 

57--- 

58 

59# あなたへの厳格な指示 

60上記の情報に基づき、以下の思考プロセスに従って、FKBエントリを生成してください。 

61 

62## 思考プロセス 

631. **根本原因の特定**: エラーログとコードを注意深く読み、バグの根本原因を特定せよ。（例：「テストコードが、ネストされた関数を直接インポートしようとしている」） 

642. **エラーシグネチャの一般化**: エラーログから、この種のエラーを将来確実に捕捉できる、汎用的な正規表現（regex）を `error_signature` として考案せよ。 

653. **完全な解決策の考案**: このバグを修正し、**テストを完全にパスさせる**には、どのファイルをどのように変更する必要があるかを具体的に考えよ。 

66 - 例えば、`ImportError`を修正する場合、`import`文の修正だけでなく、そのインポートを利用している箇所の**関数名**も修正する必要があるかもしれない。 

67 - 例えば、ソースコードの関数名を変更する場合、テストコードの呼び出し箇所も変更する必要がある。 

68 - この具体的な修正を、`llm_diagnose_and_fix`の`instruction`として記述せよ。 

694. **最終的なJSONの構築**: 上記の分析結果を基に、`id`, `error_signature`, `cause`, `target`, `solution_pattern`, `description` を持つ、単一のJSONオブジェクトを生成せよ。 

70 

71## `solution_pattern` の詳細なルール 

72- `solution_pattern` は、必ず "type" キーを持つ **JSONオブジェクト（辞書）**でなければならない。 

73- `type` が `"llm_diagnose_and_fix"` の場合: `"instruction"` キーに、**テストを完全にパスさせるための、包括的な修正指示**を記述せよ。 

74 

75## JSON出力例 

76```json 

77{{ 

78 "id": "FKB-SUGGESTION-0001", 

79 "error_signature": "ImportError: cannot import name 'add'", 

80 "cause": "テストコードが、ネストされた関数や、誤った名前の関数を直接インポートしようとしている。", 

81 "target": "test_file", 

82 "solution_pattern": {{ 

83 "type": "llm_diagnose_and_fix", 

84 "instruction": "The test failed with an ImportError. Analyze the source code to find the correct way to access the intended function. Modify the test code to import the parent function and then call it to get the nested function, or fix the imported function name if it's a simple typo. The goal is to make the test pass." 

85 }}, 

86 "description": "ネストされた関数やタイポによる不正なインポートが原因のエラーを、テストコード側を修正することで解決する知識。" 

87}} 

88``` 

89 

90# 絶対的な出力ルール 

91- **生成したJSONオブジェクトのみ**を出力すること。 

92- 説明、前置き、その他のテキストは一切含めてはならない。 

93- `id` は `"FKB-SUGGESTION-XXXX"` の形式とせよ。 

94- `cause` と `description` は、日本語で簡潔に記述すること。 

95- JSONは必ず `{{` で始まり、 `}}` で終わる単一のオブジェクトであること。 

96""" 

97 # ▲▲▲▲▲ ここまでが最重要修正点 ▲▲▲▲▲ 

98 try: 

99 response_str = self._call_llm(prompt, self.SYSTEM_PROMPT, temperature=0.3, as_json=True) 

100 match = re.search(r'\{.*\}', response_str, re.DOTALL) 

101 if not match: 

102 self.logger.error(f"LLM response does not contain a JSON object. Raw response:\n{response_str}") 

103 return None 

104 

105 json_str = match.group(0) 

106 suggestion = json.loads(json_str) 

107 self.logger.info("Successfully generated a new FKB suggestion.") 

108 return suggestion 

109 

110 except json.JSONDecodeError as e: 

111 self.logger.error(f"Invalid JSON generated by LLM: {e}. Raw response:\n{response_str}") 

112 return None 

113 except Exception as e: 

114 self.logger.error(f"An unexpected error occurred in PostmortemAgent: {e}", exc_info=True) 

115 return None
