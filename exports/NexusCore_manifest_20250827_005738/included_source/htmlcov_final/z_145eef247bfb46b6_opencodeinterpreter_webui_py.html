25def process_audio(audio_file): 

26 try: 

27 if audio_file is None: 

28 raise gr.Warning("録音がキャンセルされました。") 

29 with open(audio_file, "rb") as f: 

30 transcript = client.audio.transcriptions.create( 

31 model="whisper-1", 

32 file=f, 

33 language="ja", 

34 response_format="text" 

35 ) 

36 return transcript 

37 except gr.Warning as w: 

38 raise w 

39 except Exception as e: 

40 logging.error(f"音声処理エラー: {str(e)}") 

41 raise gr.Error(f"音声処理エラー: {e}")

44def update_uuid(dialog_info): 

45 new_uuid = str(uuid4()) 

46 logging.info(f"allocating new uuid {new_uuid} for conversation...") 

47 return [new_uuid, dialog_info[1]]

49def history_to_messages(history): 

50 messages = [] 

51 for msg in history: 

52 if isinstance(msg, dict) and "role" in msg and "content" in msg: 

53 messages.append(msg) 

54 elif isinstance(msg, (list, tuple)) and len(msg) == 2: 

55 messages.append({"role": "user", "content": msg[0]}) 

56 messages.append({"role": "assistant", "content": msg[1]}) 

57 return messages

59def bot(user_message, files, history, dialog_info, frontend_preview): 

60 try: 

61 if files is None: 

62 files = [] 

63 file_info, file_content, file_types, frontend_preview_str = handle_uploaded_files(files) 

64 user_input = user_message 

65 if file_info: 

66 user_input += "\n" + file_info 

67 if file_content: 

68 user_input += f"\n[\u30d5\u30a1\u30a4\u30eb\u5185\u5bb9（4000\u6587\u5b57\u307e\u3067）]\n{file_content[:4000]}" 

69 

70 prev_messages = history if history and isinstance(history[0], dict) else history_to_messages(history) 

71 ai_response = "ファイルまたはテキストを受け取りました。" 

72 

73 chatbot_value = prev_messages + [ 

74 {"role": "user", "content": user_message}, 

75 {"role": "assistant", "content": ai_response} 

76 ] 

77 return chatbot_value, chatbot_value, dialog_info, frontend_preview_str

79 except Exception as e:

98def gradio_launch(): 

99 with gr.Blocks() as demo: 

100 with gr.Tabs(): 

101 # タブ1: 修正・テスト 

102 with gr.Tab("🛠 修正サイクル"): 

103 code_input = gr.Textbox(label="💡 入力コード（エラーあり可）", lines=10) 

104 btn_testgen = gr.Button("🧪 ユニットテスト生成") 

105 test_output = gr.Code(label="📄 生成されたユニットテスト") 

106 btn_run_repair = gr.Button("🔁 修正のみ実行") 

107 btn_run_test_repair = gr.Button("🧪 修正+\u30c6スト一括") 

108 output_code = gr.Code(label="✅ 修正済みコード or レポート") 

109 

110 btn_testgen.click(fn=generate_and_show_tests, inputs=code_input, outputs=test_output) 

111 btn_run_repair.click(fn=run_and_repair, inputs=code_input, outputs=output_code) 

112 btn_run_test_repair.click(fn=run_test_and_repair, inputs=code_input, outputs=output_code) 

113 

114 # タブ2: チャット＋ファイル分析 

115 with gr.Tab("💬 Chat + ファイル分析"): 

116 chatbot = gr.Chatbot(label="OpenCodeInterpreter", height=600, type="messages") 

117 msg = gr.Textbox(placeholder="メッセージ入力 or 音声録音", scale=5) 

118 file_input = gr.File(file_types=[".py", ".txt", ".md", ".json", ".zip"], file_count="multiple") 

119 file_list = gr.Textbox(label="アップロードファイル一覧", interactive=False, max_lines=10) 

120 audio_input = gr.Audio(sources="microphone", type="filepath", label="音声録音") 

121 frontend_preview = gr.Textbox(label="ファイル先頭プレビュー（100字）") 

122 submit = gr.Button("Submit") 

123 clear = gr.Button("Clear") 

124 download_btn = gr.DownloadButton("履歴ダウンロード") 

125 session_state = gr.State([]) 

126 dialog_info = gr.State(["", 0]) 

127 

128 demo.load(update_uuid, dialog_info, dialog_info) 

129 file_input.change(file_list_display, inputs=file_input, outputs=file_list) 

130 audio_input.change(process_audio, inputs=audio_input, outputs=msg) 

131 submit.click(bot, [msg, file_input, session_state, dialog_info, frontend_preview], [chatbot, session_state, dialog_info, frontend_preview]) 

132 clear.click(lambda h, d: ([], [], update_uuid(d), ""), [session_state, dialog_info], [chatbot, session_state, dialog_info, frontend_preview]) 

133 download_btn.click(download_history, [session_state], download_btn) 

134 

135 demo.queue(max_size=20) 

136 demo.launch(share=True, inbrowser=True)
