79# === GPT呼び出しとコード抽出 === 

80def extract_code_and_reason(full_response): 

81 code_match = re.search(r"```(?:python)?\n(.*?)```", full_response, re.DOTALL) 

82 reason_match = re.split(r"```.*?```", full_response, maxsplit=1) 

83 code = code_match.group(1).strip() if code_match else "" 

84 reason = reason_match[1].strip() if len(reason_match) > 1 else "" 

85 return code, reason

87def call_gpt(prompt): 

88 response = client.chat.completions.create( 

89 model="gpt-4", 

90 messages=[{"role": "user", "content": prompt}], 

91 temperature=0 

92 ) 

93 return response.choices[0].message.content.strip()
