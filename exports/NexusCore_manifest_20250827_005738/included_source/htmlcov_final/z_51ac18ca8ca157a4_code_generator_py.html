3from openai import OpenAI 

4import os 

5from dotenv import load_dotenv

7# .envからAPIキーを読み込む 

8load_dotenv() 

9client = OpenAI(api_key=os.getenv("OPENAI_API_KEY")) 

10 

11def generate_code_from_text(natural_text: str) -> str: 

12 """ 

13 自然言語から関数名付きのPythonコードをGPTで生成 

14 

15 Parameters: 

16 natural_text (str): ユーザーの意図や実装希望内容の自然文 

17 

18 Returns: 

19 str: GPTによるPythonコード（コードブロック付き） 

20 """ 

21 prompt = f""" 

22以下の説明に基づいて、関数名・ドキュメント付きのPython関数コードを生成してください。 

23 

24【説明】{natural_text} 

25 

26# 出力フォーマット： 

27- コードブロック内に、コメント・関数定義・処理本体を含める 

28- print()などの使用例があれば最後に追加してもよい 

29- コード以外の文章は一切含めない 

30 

31""" 

32 try: 

33 response = client.chat.completions.create( 

34 model="gpt-4", 

35 messages=[{"role": "user", "content": prompt}], 

36 temperature=0.2 

37 ) 

38 return response.choices[0].message.content.strip() 

39 except Exception as e: 

40 return f"⚠️ GPT code generation failed: {e}"
