19class KnowledgeCuratorAgent: 

20 def __init__(self, api_key: str, model: str): 

21 self.logger = logging.getLogger(self.__class__.__name__) 

22 self.api_key = api_key 

23 self.model = model 

24 

25 def validate_fkb_suggestion( 

26 self, 

27 suggestion: dict, 

28 original_project_path: str, 

29 failed_test_path: str, 

30 related_source_path: str, 

31 # ▼▼▼▼▼ 【最重要修正点】生のテスト失敗ログを追加 ▼▼▼▼▼ 

32 original_test_output: str 

33 # ▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲ 

34 ) -> bool: 

35 """ 

36 提案されたFKBエントリを一時的なサンドボックス環境で検証する。 

37 """ 

38 self.logger.info(f"Starting validation for FKB suggestion: {suggestion.get('id', 'N/A')}") 

39 

40 with tempfile.TemporaryDirectory(prefix="k_curator_") as sandbox_path: 

41 try: 

42 # 1. プロジェクトの関連ファイルのみをサンドボックスにコピー 

43 self.logger.debug(f"Creating minimal sandbox at: {sandbox_path}") 

44 

45 # 必要なディレクトリ構造を作成 

46 Path(sandbox_path, os.path.dirname(os.path.relpath(related_source_path, original_project_path))).mkdir(parents=True, exist_ok=True) 

47 Path(sandbox_path, os.path.dirname(os.path.relpath(failed_test_path, original_project_path))).mkdir(parents=True, exist_ok=True) 

48 

49 # 関連ファイルのみをコピー 

50 shutil.copy(related_source_path, os.path.join(sandbox_path, os.path.relpath(related_source_path, original_project_path))) 

51 shutil.copy(failed_test_path, os.path.join(sandbox_path, os.path.relpath(failed_test_path, original_project_path))) 

52 

53 # __init__.pyを作成 

54 for dirpath, _, _ in os.walk(sandbox_path): 

55 Path(dirpath, "__init__.py").touch() 

56 

57 # 2. 一時的なFKBを作成 

58 temp_fkb_path = os.path.join(sandbox_path, "temp_fkb.json") 

59 with open(temp_fkb_path, 'w', encoding='utf-8') as f: 

60 json.dump([suggestion], f, ensure_ascii=False, indent=2) 

61 self.logger.debug("Created temporary FKB with new suggestion.") 

62 

63 # 3. サンドボックス内でDebuggerAgentを初期化 

64 debugger = DebuggerAgent(self.api_key, self.model, knowledge_base_path=temp_fkb_path, project_path=sandbox_path) 

65 patcher = PatchApplier() 

66 

67 # 4. 自己修復を試行 

68 # ▼▼▼▼▼ 【最重要修正点】サンドボックス内でテストを再実行するのではなく、元のエラーログを使用 ▼▼▼▼▼ 

69 files_context = { 

70 "source_file": os.path.join(sandbox_path, os.path.relpath(related_source_path, original_project_path)), 

71 "test_file": os.path.join(sandbox_path, os.path.relpath(failed_test_path, original_project_path)) 

72 } 

73 debug_result = debugger.debug(original_test_output, files_context) 

74 # ▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲ 

75 

76 if not (debug_result and debug_result.get("patch")): 

77 self.logger.warning("Validation failed: Debugger did not generate a patch with the new knowledge.") 

78 return False 

79 

80 # 5. パッチを適用し、テストが成功するか確認 

81 was_applied = patcher.apply(debug_result["patch"], sandbox_path) 

82 if not was_applied: 

83 self.logger.warning("Validation failed: Generated patch could not be applied.") 

84 return False 

85 

86 tests_passed, _ = self._run_tests_in_sandbox(sandbox_path, os.path.relpath(failed_test_path, original_project_path)) 

87 if tests_passed: 

88 self.logger.info("✅ Validation successful! The new knowledge correctly fixed the bug.") 

89 return True 

90 else: 

91 self.logger.warning("Validation failed: Tests still fail after applying the patch.") 

92 return False 

93 

94 except Exception as e: 

95 self.logger.error(f"An error occurred during validation: {e}", exc_info=True) 

96 return False

98 def _run_tests_in_sandbox(self, sandbox_path: str, test_file_rel_path: str) -> tuple[bool, str]: 

99 """サンドボックス内でpytestを実行する""" 

100 result = subprocess.run( 

101 [sys.executable, "-m", "pytest", test_file_rel_path], 

102 cwd=sandbox_path, 

103 capture_output=True, text=True, encoding='utf-8', errors='replace' 

104 ) 

105 output = result.stdout + "\n" + result.stderr 

106 return result.returncode == 0, output
