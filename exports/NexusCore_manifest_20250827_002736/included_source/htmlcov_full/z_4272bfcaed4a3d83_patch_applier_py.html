2# フォルダ: src/agents 

3# ファイル名: patch_applier.py 

4# メモ: 堅牢性と信頼性を最大化するため、専門ライブラリ`patch`を 

5# 正しく、安全に利用する完成版。 

6# ============================================================================== 

7import logging 

8import os 

9import patch # 専門ライブラリをインポート 

10 

11class PatchApplier: 

12 """ 

13 'unified diff' 形式のパッチをソースコードファイルに適用するためのクラス。 

14 専門ライブラリ`patch`を利用し、複雑なパッチも正確に適用する。 

15 """ 

16 

17 def apply(self, patch_str: str, project_path: str) -> bool: 

18 """ 

19 指定されたプロジェクトパスを基準にパッチを適用します。 

20 

21 Args: 

22 patch_str (str): unified diff形式のパッチ文字列。 

23 project_path (str): パッチを適用するプロジェクトのルートパス。 

24 

25 Returns: 

26 bool: パッチの適用が成功した場合はTrue、失敗した場合はFalse。 

27 """ 

28 if not patch_str: 

29 logging.warning("Patch is empty. Nothing to apply.") 

30 return False 

31 

32 try: 

33 # パッチ文字列からパッチセットを解析 

34 patch_set = patch.fromstring(patch_str.encode('utf-8')) 

35 

36 # ★ 解析失敗時のエラーハンドリングを追加 

37 if not patch_set: 

38 logging.error("Failed to parse patch string. It might be invalid or malformed.") 

39 logging.debug(f"Invalid patch string:\n{patch_str}") 

40 return False 

41 

42 # ★ パッチを適用する基準ディレクトリ(root)を、プロジェクトのパスに正しく設定 

43 success = patch_set.apply(root=project_path) 

44 

45 if success: 

46 logging.info(f"Successfully applied patch within project: {project_path}") 

47 return True 

48 else: 

49 logging.error(f"Failed to apply patch within project: {project_path}. The patch may be invalid or already applied.") 

50 return False 

51 except Exception as e: 

52 logging.error(f"An error occurred while applying patch with 'patch' library: {e}", exc_info=True) 

53 return False
