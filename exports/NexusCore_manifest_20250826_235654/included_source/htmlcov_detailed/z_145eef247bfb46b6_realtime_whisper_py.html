27def record_and_process_audio(max_duration=60, sample_rate=16000): 

28 print(f"録音開始: 最大{max_duration}秒、エンターキーで終了") 

29 recording = [] 

30 event = threading.Event() 

31 start_time = time.time() 

32 

33 def callback(indata, frames, t, status): 

34 if time.time() - start_time > max_duration: 

35 event.set() 

36 raise sd.CallbackAbort 

37 recording.append(indata.copy()) 

38 

39 def record_thread(): 

40 with sd.InputStream(samplerate=sample_rate, channels=1, callback=callback): 

41 event.wait() 

42 

43 def key_thread(): 

44 input() # エンターキー待ち 

45 event.set() 

46 

47 t1 = threading.Thread(target=record_thread) 

48 t2 = threading.Thread(target=key_thread) 

49 t1.start() 

50 t2.start() 

51 t2.join(timeout=max_duration) 

52 t1.join(timeout=1) 

53 

54 if not recording: 

55 return None 

56 

57 audio_np = np.concatenate(recording, axis=0).flatten() 

58 

59 # 一時ファイルに保存 

60 temp_wav = tempfile.NamedTemporaryFile(suffix='.wav', delete=False) 

61 write(temp_wav.name, sample_rate, audio_np) 

62 

63 # ノイズ除去・音量正規化処理 

64 y, sr = librosa.load(temp_wav.name, sr=sample_rate) 

65 noise_sample = y[:int(sr*0.5)] # 最初の0.5秒をノイズと仮定 

66 y_denoised = nr.reduce_noise(y=y, y_noise=noise_sample, sr=sr, stationary=False) 

67 y_normalized = librosa.util.normalize(y_denoised) 

68 

69 # 処理後の音声を別ファイルに保存 

70 processed_wav = tempfile.NamedTemporaryFile(suffix='_processed.wav', delete=False) 

71 sf.write(processed_wav.name, y_normalized, sr) 

72 

73 # 元の録音ファイルは削除 

74 temp_wav.close() 

75 os.unlink(temp_wav.name) 

76 

77 return processed_wav.name

79def transcribe_with_whisper(audio_path): 

80 with open(audio_path, "rb") as f: 

81 transcript = openai.audio.transcriptions.create(

83 file=f, 

84 language="ja", 

85 response_format="text" 

86 ) 

87 return transcript
