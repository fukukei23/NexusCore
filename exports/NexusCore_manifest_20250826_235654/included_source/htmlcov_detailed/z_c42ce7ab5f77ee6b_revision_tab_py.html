22def save_file(path, content): 

23 with open(path, "w", encoding="utf-8") as f: 

24 f.write(content)

26def read_file(path): 

27 with open(path, "r", encoding="utf-8") as f: 

28 return f.read()

30def run_pytest(): 

31 try: 

32 result = subprocess.run( 

33 ["pytest", TEST_FILE], 

34 stdout=subprocess.PIPE, 

35 stderr=subprocess.PIPE, 

36 text=True 

37 ) 

38 output = result.stdout + "\n" + result.stderr 

39 save_file(RESULT_LOG, output)

44def generate_prompt(main_file, related_files, version_summary, history_summary, failed_tests, user_instruction): 

45 return f""" 

46【前提】 

47- 対象ファイル: {main_file} 

48- 関連ファイル・依存関係: {related_files} 

49- 現在のバージョン: {version_summary} 

50- 修正履歴: {history_summary} 

51- 直近のテスト失敗内容: {failed_tests} 

52- ユーザーからの追加指示: {user_instruction} 

53 

54【タスク】 

551. 修正版を提案してください。 

562. 修正理由・依存関係・テスト失敗の考察を記述してください。 

573. 出力は必ずコードブロック形式で。 

58"""

60def extract_code_and_reason(full_response): 

61 code_match = re.search(r"```(?:python)?\n(.*?)```", full_response, re.DOTALL) 

62 reason_match = re.split(r"```.*?```", full_response, maxsplit=1) 

63 code = code_match.group(1).strip() if code_match else "" 

64 reason = reason_match[1].strip() if len(reason_match) > 1 else "" 

65 return code, reason

67def call_gpt(prompt): 

68 response = client.chat.completions.create( 

69 model="gpt-4", 

70 messages=[{"role": "user", "content": prompt}], 

71 temperature=0 

72 ) 

73 return response.choices[0].message.content.strip()

75def save_patch_history(code, reason, prompt): 

76 now = datetime.now().strftime("%Y%m%d_%H%M%S") 

77 data = { 

78 "timestamp": now, 

79 "code": code, 

80 "reason": reason,

84 save_file(os.path.join(HISTORY_DIR, f"patch_{now}.json"), json.dumps(data, indent=2, ensure_ascii=False))

101 def generate_revision(user_code, user_note, fail_log):
