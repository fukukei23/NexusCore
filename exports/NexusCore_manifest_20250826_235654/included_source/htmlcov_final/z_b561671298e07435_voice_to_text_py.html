41def record_until_keypress(max_duration=60, sample_rate=16000): 

42 """ 

43 エンターキーが押されるか、最大max_duration秒まで録音 

44 """ 

45 print(f"録音中... 最大{max_duration}秒、エンターキーで終了します") 

46 recording = [] 

47 event = threading.Event() 

48 start_time = time.time() 

49 

50 def callback(indata, frames, t, status): 

51 # 最大時間を超えたら録音停止 

52 if time.time() - start_time > max_duration: 

53 event.set() 

54 raise sd.CallbackAbort 

55 recording.append(indata.copy()) 

56 

57 def record_thread(): 

58 with sd.InputStream(samplerate=sample_rate, channels=1, callback=callback): 

59 event.wait() 

60 

61 def key_thread(): 

62 input() # エンターキー待ち 

63 event.set() 

64 

65 t1 = threading.Thread(target=record_thread) 

66 t2 = threading.Thread(target=key_thread) 

67 t1.start() 

68 t2.start() 

69 t2.join(timeout=max_duration) 

70 t1.join(timeout=1) 

71 

72 if recording: 

73 return np.concatenate(recording, axis=0), sample_rate 

74 return None, sample_rate

76def transcribe_with_whisper(audio_path): 

77 """Whisper APIで音声ファイルを文字起こし""" 

78 with open(audio_path, "rb") as f: 

79 transcript = openai.audio.transcriptions.create( 

80 model="whisper-1", 

81 file=f, 

82 language="ja", 

83 response_format="text" 

84 ) 

85 return transcript
